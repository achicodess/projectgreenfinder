  /* ═══ Falling Leaves (Canvas) ═══ */
  (function () {
    const canvas = document.getElementById('leaf-canvas');
    const ctx = canvas.getContext('2d');
    let W, H;

    // Leaf shape definitions as Path2D-compatible SVG path strings (centered at 0,0)
    const shapeDefs = [
      'M0,-28 C12,-20 18,-6 16,10 C14,24 -14,24 -16,10 C-18,-6 -12,-20 0,-28Z',
      'M0,-32 C8,-18 10,0 8,18 C5,30 -5,30 -8,18 C-10,0 -8,-18 0,-32Z',
      'M0,-20 C14,-14 18,2 14,16 C10,26 -10,26 -14,16 C-18,2 -14,-14 0,-20Z',
      'M0,-18 C6,-26 20,-22 20,-10 C20,4 8,18 0,26 C-8,18 -20,4 -20,-10 C-20,-22 -6,-26 0,-18Z',
    ];
    const paths = shapeDefs.map(d => new Path2D(d));

    const colors = ['#4a8055','#5c9e6b','#3a6845','#7dba8a','#a8d4ac','#c8a845','#2d6040','#6aae78'];

    function resize() {
      W = canvas.width  = window.innerWidth;
      H = canvas.height = window.innerHeight;
    }

    function makeLeaf(immediate) {
      const onLeft = Math.random() < 0.5;
      return {
        x: onLeft ? Math.random() * W * 0.17 : W * 0.83 + Math.random() * W * 0.17,
        y: immediate ? Math.random() * H : -70,
        vx: (Math.random() - 0.5) * 0.35,
        vy: 0.45 + Math.random() * 0.85,
        angle: Math.random() * Math.PI * 2,
        va:   (Math.random() - 0.5) * 0.022,
        scale: 0.5 + Math.random() * 0.65,
        shape: Math.floor(Math.random() * paths.length),
        color: colors[Math.floor(Math.random() * colors.length)],
        opacity: 0.25 + Math.random() * 0.38,
        t: Math.random() * Math.PI * 2,
        swaySpeed: 0.007 + Math.random() * 0.011,
        swayAmp:   0.28 + Math.random() * 0.5,
      };
    }

    const leaves = Array.from({ length: 24 }, () => makeLeaf(true));

    function frame() {
      ctx.clearRect(0, 0, W, H);
      for (const l of leaves) {
        l.t += l.swaySpeed;
        l.x += l.vx + Math.sin(l.t) * l.swayAmp;
        l.y += l.vy;
        l.angle += l.va;
        if (l.y > H + 90) Object.assign(l, makeLeaf(false));

        ctx.save();
        ctx.translate(l.x, l.y);
        ctx.rotate(l.angle);
        ctx.scale(l.scale, l.scale);
        ctx.globalAlpha = l.opacity;
        ctx.fillStyle = l.color;
        ctx.fill(paths[l.shape]);
        // midrib line
        ctx.strokeStyle = 'rgba(0,0,0,0.18)';
        ctx.lineWidth = 0.8;
        ctx.beginPath();
        ctx.moveTo(0, -22); ctx.lineTo(0, 22);
        ctx.stroke();
        ctx.restore();
      }
      requestAnimationFrame(frame);
    }

    resize();
    window.addEventListener('resize', resize);
    frame();
  })();

  document.querySelectorAll('.btn').forEach(btn => {
    btn.addEventListener('mousemove', e => {
      const r = btn.getBoundingClientRect();
      btn.style.setProperty('--rx', ((e.clientX - r.left) / r.width  * 100) + '%');
      btn.style.setProperty('--ry', ((e.clientY - r.top)  / r.height * 100) + '%');
    });
  });

  /* ═══ App ═══ */
  let model, webcam, maxPredictions;
  let webcamRunning = false;
  let frameCount = 0, fpsTimer = null, lastFpsSnap = performance.now();

  const $ = id => document.getElementById(id);

  function setModel(ok, txt) {
    $('f-model').className = 'f-dot' + (ok ? ' on' : '');
    $('f-model-txt').textContent = txt;
  }
  function setCam(state, txt) {
    $('f-cam').className = 'f-dot' + (state === 'live' ? ' live' : '');
    $('f-cam-txt').textContent = txt;
  }
  function showLoading(msg) {
    $('loading-text').textContent = msg;
    $('loading-wrap').classList.add('active');
    $('empty-state').classList.add('hidden');
    $('results-body').style.display = 'none';
  }
  function hideLoading() { $('loading-wrap').classList.remove('active'); }
  function toast(msg) {
    const el = $('toast');
    el.textContent = '⚠  ' + msg;
    el.classList.add('show');
    setTimeout(() => el.classList.remove('show'), 4200);
  }

  async function ensureModel() {
    if (model) return;
    setModel(false, 'Loading model…');
    model = await tmImage.load('./model.json', './metadata.json');
    maxPredictions = model.getTotalClasses();
    setModel(true, `Model ready · ${maxPredictions} plants`);
  }

  async function startWebcam() {
    try {
      showLoading('Waking the lens…');
      await ensureModel();
      $('loading-text').textContent = 'Starting camera…';
      webcam = new tmImage.Webcam(320, 320, true);
      await webcam.setup();
      await webcam.play();
      $('webcam-mount').innerHTML = '';
      $('webcam-mount').appendChild(webcam.canvas);
      $('webcam-container').classList.add('active');
      $('btn-webcam').style.display = 'none';
      $('btn-stop').style.display = '';
      setCam('live', 'Camera live');
      $('f-fps').style.display = '';
      webcamRunning = true;
      buildRows(maxPredictions);
      hideLoading();
      fpsTimer = setInterval(() => {
        const now = performance.now();
        $('f-fps-txt').textContent = Math.round(frameCount / ((now - lastFpsSnap) / 1000)) + ' fps';
        frameCount = 0; lastFpsSnap = now;
      }, 1000);
      loop();
    } catch (e) {
      hideLoading();
      toast('Camera error: ' + (e.message || e));
    }
  }

  function stopWebcam() {
    if (webcam) { webcam.stop(); webcam = null; }
    webcamRunning = false;
    $('webcam-container').classList.remove('active');
    $('btn-webcam').style.display = '';
    $('btn-stop').style.display = 'none';
    setCam('off', 'Camera off');
    $('f-fps').style.display = 'none';
    clearInterval(fpsTimer);
  }

  async function loop() {
    if (!webcamRunning) return;
    webcam.update(); frameCount++;
    const preds = await model.predict(webcam.canvas);
    renderPreds(preds);
    requestAnimationFrame(loop);
  }

  function buildRows(n) {
    const lc = $('label-container');
    lc.innerHTML = '';
    for (let i = 0; i < n; i++) {
      const div = document.createElement('div');
      div.className = 'pred-item';
      div.style.animationDelay = (i * 0.04) + 's';
      div.innerHTML = `
        <div class="pred-fill-bg"></div>
        <div class="pred-inner">
          <div class="pred-left">
            <span class="pred-num">${i + 1}</span>
            <div>
              <span class="pred-name">—</span>
              <span class="pred-sublabel">awaiting scan</span>
            </div>
          </div>
          <div class="pred-right">
            <span class="best-tag">Best Match</span>
            <span class="pred-pct">—</span>
          </div>
        </div>
        <div class="pred-track"><div class="pred-bar" style="width:0%"></div></div>`;
      lc.appendChild(div);
    }
    $('empty-state').classList.add('hidden');
    $('results-body').style.display = '';
    $('results-meta').textContent = n + ' species';
  }

  function renderPreds(preds) {
    const sorted = [...preds].sort((a, b) => b.probability - a.probability);
    const items  = $('label-container').querySelectorAll('.pred-item');
    sorted.forEach((p, i) => {
      const el = items[i]; if (!el) return;
      const pct = (p.probability * 100).toFixed(1);
      el.querySelector('.pred-name').textContent     = p.className;
      el.querySelector('.pred-sublabel').textContent = i === 0 ? 'top identification' : 'possible match';
      el.querySelector('.pred-pct').textContent      = pct + '%';
      el.querySelector('.pred-bar').style.width      = pct + '%';
      el.querySelector('.pred-fill-bg').style.width  = pct + '%';
      el.querySelector('.pred-num').textContent      = i + 1;
      el.classList.toggle('champion', i === 0);
    });
  }

  function triggerUpload()   { $('image-upload').click(); }
  function handleZoneClick() { if (!$('preview-wrap').classList.contains('visible')) triggerUpload(); }
  function onDragOver(e)     { e.preventDefault(); $('drop-zone').classList.add('drag-over'); }
  function onDragLeave()     { $('drop-zone').classList.remove('drag-over'); }
  function onDrop(e) {
    e.preventDefault(); $('drop-zone').classList.remove('drag-over');
    const f = e.dataTransfer?.files?.[0];
    if (f?.type.startsWith('image/')) processFile(f);
    else toast('Please drop a valid image file.');
  }
  function handleFileInput(input) { if (input.files?.[0]) processFile(input.files[0]); }

  async function processFile(file) {
    try {
      const url = URL.createObjectURL(file);
      $('preview-img').src = url;
      $('drop-hint').style.display = 'none';
      $('preview-wrap').classList.add('visible');
      $('drop-zone').classList.add('has-image');
      if (webcamRunning) stopWebcam();
      showLoading('Reading the leaves…');
      await ensureModel();
      const img = new Image();
      img.src = url;
      await new Promise((res, rej) => { img.onload = res; img.onerror = rej; });
      const preds = await model.predict(img);
      buildRows(preds.length);
      renderPreds(preds);
      hideLoading();
    } catch (e) {
      hideLoading();
      toast('Could not read this image: ' + (e.message || e));
    }
  }
